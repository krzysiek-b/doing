def add(note, replace: false)
  clear if replace
  case note
  when String
    append_string(note)
  when Array
    append(note)
  end
end

  to_s == other.to_s
end

def to_s(prefix: "\t\t")
  compress.strip_lines.map { |l| "#{prefix}#{l}" }.join("\n")
end
